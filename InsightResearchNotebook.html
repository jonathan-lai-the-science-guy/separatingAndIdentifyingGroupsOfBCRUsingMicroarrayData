import gc

def loadData(download_from_online=False):
    gse = None
    if(download_from_online):
        gse = GEOparse.get_GEO(geo="GSE20194", destdir="./")
    else:
        gse = GEOparse.get_GEO(filepath="./GSE20194_family.soft.gz",silent=True)

    pivot_samples = gse.pivot_samples('VALUE')

    gc.collect()
    return(pivot_samples,gse)

def parseMetadata(gse):
    TABLE = gse.gpls['GPL96'].table

    def parseGeneOntologyString(inputString):
        arrayOfStringSegments = inputString.split("//")
        return arrayOfStringSegments[1].strip()

    def parseGeneTitleString(inputString):
        arrayOfStringSegments = inputString.split("///")
        if("microRNA" in arrayOfStringSegments[0] and\
           len(arrayOfStringSegments) > 2):
            return arrayOfStringSegments[1].strip()
        return arrayOfStringSegments[0].strip()
    
    gene_ontology_biological_process = {}
    biological_process_count = {}
    for i,j,k,l in zip(TABLE["ID"],\
                       TABLE["Gene Ontology Biological Process"],\
                     TABLE["Gene Ontology Molecular Function"],\
                      TABLE["Gene Title"]):
        biological_process = "Biological process not-annotated"
        if(isinstance(l, str)):
            biological_process = parseGeneTitleString(l)
        elif(isinstance(j, str)):
            biological_process = parseGeneOntologyString(j)
        elif(isinstance(k,str)):
            biological_process = parseGeneOntologyString(k)

        gene_ontology_biological_process[i.__str__()] = biological_process
        if(biological_process in biological_process_count):
            biological_process_count[biological_process] += 1
        else:
            biological_process_count[biological_process] = 1

    gc.collect()
    return biological_process_count,gene_ontology_biological_process

BPC,GOBP = parseMetadata(gse)
